== `@Retention`

=== `@Retention`

[.fragment]
[source, java]
----
@Documented
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.ANNOTATION_TYPE)
public @interface Retention {
    RetentionPolicy value();
}
----

=== `RetentionPolicy`

[.step]
* `SOURCE`
* `CLASS`
* `RUNTIME`

== `@Target`

=== `@Target`

[.fragment]
[source, java]
----
@Documented
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.ANNOTATION_TYPE)
public @interface Target {
    ElementType[] value();
}
----

=== `ElementType`

[.step]
* `TYPE`
* `FIELD`
* `METHOD`
* `PARAMETER`
* `CONSTRUCTOR`
* `LOCAL_VARIABLE`

=== `ElementType`

[.step]
* `ANNOTATION_TYPE`
* `PACKAGE`
* `TYPE_PARAMETER`
* `TYPE_USE`
* `MODULE`

== Definition Annotation

=== Creating Custom Annotation

[.fragment]
[source,java]
----
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.METHOD)
@interface MyAnnotation {
    String str();
    int val();
}
----

=== Using Annotation

[.fragment]
[source,java]
----
@MyAnnotation(str = "Пример аннотации", val = 988)
public static void myMethod() {
    // code
}
----
